package com.example;

import com.example.api.ElpriserAPI;

import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.List;

public class Main {
    public static void main(String[] args) {

        String zone = "SE3";
        LocalDate date = LocalDate.now();

        for (int i = 0; i < args.length; i++) {
            switch (args[i]) {
                case "--zone":
                    if (i + 1 < args.length) {
                        zone = args[i + 1].toUpperCase();
                        i++;
                    }
                    break;
                case "--date":
                    if (i + 1 < args.length) {
                        try {
                            date = LocalDate.parse(args[i + 1]);
                            i++;
                        } catch (DateTimeParseException e) {
                            System.out.println("MM-dd.");
                            return;
                        }
                    }
            }
        }

        // api o prisklass
        ElpriserAPI elpriserAPI = new ElpriserAPI();
        ElpriserAPI.Prisklass prisklass;
        try {
            prisklass = ElpriserAPI.Prisklass.valueOf(zone);
        } catch (IllegalArgumentException e) {
            System.out.println("Felaktig zon: " + zone + ". Använder SE3.");
            prisklass = ElpriserAPI.Prisklass.SE3;  // fallback till SE3
        }

        // hämtat priser
        List<ElpriserAPI.Elpris> prices = elpriserAPI.getPriser(date, prisklass);

        if (prices.isEmpty()) {
            System.out.println("Inga priser för " + date + " i zon " + zone + ".");
            return;
        }

        // billigast, dyrast o medelpris
        ElpriserAPI.Elpris billigast = prices.get(0);
        ElpriserAPI.Elpris dyrast = prices.get(0);
        double sum = 0;

        for (ElpriserAPI.Elpris pris : prices) {
            double currentPrice = pris.sekPerKWh();
            sum += currentPrice;

            if (currentPrice < billigast.sekPerKWh()) {
                billigast = pris;
            }
            if (currentPrice > dyrast.sekPerKWh()) {
                dyrast = pris;
            }
        }

        double medelpris = sum / prices.size();
        DateTimeFormatter timeFormat = DateTimeFormatter.ofPattern("HH:mm");

        System.out.printf("Medelpris för %s i zon %s: %.4f SEK/kWh%n", date, zone, medelpris);
        System.out.printf("Billigaste timme: %s - %.4f SEK/kWh%n",
                billigast.timeStart().format(timeFormat), billigast.sekPerKWh());
        System.out.printf("Dyraste timme: %s - %.4f SEK/kWh%n",
                dyrast.timeStart().format(timeFormat), dyrast.sekPerKWh());

        // sliding window för opt laddningsfönster
        int[] durations = {2, 4, 8};

        for (int duration : durations) {
            if (prices.size() < duration) continue;

            double minSum = Double.MAX_VALUE;
            int minIndex = 0;

            for (int i = 0; i <= prices.size() - duration; i++) {
                double windowSum = 0;
                for (int j = i; j < i + duration; j++) {
                    windowSum += prices.get(j).sekPerKWh();
                }
                if (windowSum < minSum) {
                    minSum = windowSum;
                    minIndex = i;
                }
            }

            String startTime = prices.get(minIndex).timeStart().format(timeFormat);
            String endTime = prices.get(minIndex + duration - 1).timeStart().plusHours(1).format(timeFormat);

            System.out.printf("Bästa %dh laddningsfönster: %s - %s, Totalpris: %.4f SEK%n",
                    duration, startTime, endTime, minSum);


        }
    }
}
